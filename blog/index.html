const RICK: usize = 4

;   use       std::io::{Write, stdout}
;   use crossterm::terminal::{self, Clear, ClearType, ScrollUp}
;   use crossterm::cursor::{MoveTo}
;   use crossterm::{QueueableCommand, execute, ExecutableCommand}
;   use crossterm::event::{read, Event, KeyCode, KeyModifiers, EnableMouseCapture, DisableMouseCapture, MouseEventKind, MouseButton}
;   use       std::thread
;   use std::time::Duration
;

fn comment(coords: (u16, u16), index: usize, mark: usize, stream: &mut std::io::Stdout) -> isize
{
    stream.execute(MoveTo(coords.0 / 2 - QUESTIONS[index][mark + 5].len() as u16 / 2, coords.1 / 2))
        .unwrap()
        .write(QUESTIONS[index][mark + 5].as_bytes())
        .unwrap();
    stream.flush();

    match mark
    {
        0 => return SCORES[index].0,
        1 => return SCORES[index].1,
        2 => return SCORES[index].2,
        3 => return SCORES[index].3,
        _ => return -1,
    }
}

fn brian(index: usize, stream: &mut std::io::Stdout) -> (u16, u16)
{
    let (cols, rows) = terminal::size().unwrap();
    let curr = QUESTIONS[index];
    let kitten = cols - curr[0].len() as u16;

    if kitten > 0
    {
        execute!(stream, Clear(ClearType::All), MoveTo( kitten / 2, (rows - 14) / 2),).unwrap();
        stream.write(curr[0].as_bytes());

        stream.execute(MoveTo(kitten / 2, (rows -10) / 2))
            .unwrap()
            .write(curr[1].as_bytes()).unwrap();

        stream.execute(MoveTo(kitten / 2, (rows - 8) / 2))
            .unwrap()
            .write(curr[2].as_bytes()).unwrap();

        stream.execute(MoveTo(kitten / 2, (rows - 6) / 2))
            .unwrap()
            .write(curr[3].as_bytes()).unwrap();

        stream.execute(MoveTo(kitten / 2, (rows - 4) / 2))
            .unwrap()
            .write(curr[4].as_bytes()).unwrap();

        stream.flush();
        return (cols, rows);
    }
    else
    {
        (1, 120)
    }
}

fn main()
{
    let  mut stream = stdout();
    let (mut index, mut score, mut mark, mut offline) =(0, 0, 0, false);
    let  mut screen = brian(index, &mut stream);

    terminal::enable_raw_mode();
    execute!(stream, EnableMouseCapture).unwrap();

    while !offline
    {
        if index == RICK - 1
        {
            offline = true;
        }
        match read().unwrap()
        {
            Event::Mouse(event) =>
            {
                if event.column > 10 && event.row > 10 && event.kind == MouseEventKind::Down(MouseButton::Left)
                {
                    println!("You did it!");
                }
            },
            Event::Key(event) =>
            {
                match event.code
                {
                    KeyCode::Char('q') | KeyCode::Esc
                    => offline = true,
                    KeyCode::Char('y') | KeyCode::Char('1') | KeyCode::Up    => { mark = 0 },

                    KeyCode::Char('f') | KeyCode::Char('2') | KeyCode::Left  => { mark = 1 },

                    KeyCode::Char('g') | KeyCode::Char('3') | KeyCode::Right => { mark = 2 },

                    KeyCode::Char('v') | KeyCode::Char('4') | KeyCode::Down  => { mark = 3 },

                    KeyCode::Enter | KeyCode::Pause
                    =>{
                        score += comment(screen, index, mark, &mut stream);
                        index += 1;
                        thread::sleep(Duration::from_secs(2));
                        brian(index, &mut stream);
                    },
                    KeyCode::Tab | KeyCode::Char('e')
                    => {
                        index -= 1;
                        score -= comment(screen, index, mark, &mut stream);
                        brian(index, &mut stream);
                    },  _ => {},
                }
            },
            _ => {},
        }
    }

    execute!(stream, DisableMouseCapture).unwrap();
}
